Future<List<DiscoveredService>?> findServices(String deviceId) async {
  try {
    final result = await flutterReactiveBle.discoverServices(deviceId);
    return result;
  } catch (e) {
    if (kDebugMode) {
      print('error occurred: $e');
    }
  }
  return null;
}
